import re

text = """
Enter your English text here.
This is a sample text for testing the code. We will check if it correctly extracts all unique words from the text.
"""

# Remove punctuation, non-alphabetic characters and convert to lowercase
words = re.findall(r'\b[a-z']+\b', re.sub(r'[^\w\s]', '', text.lower()))

# Print unique words
print(sorted(set(words)))
